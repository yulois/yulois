Categoria:
  options:
    table: categorias
    timestampable: true
    sluggable: [title]
  fields:
    id					: {type: primary, strategy: identity}
    title				: {type: title, length: 255, notnull: true}

Telefono:
  options:
    table: telefonos
    timestampable: true
  fields:
    id					: {type: primary, strategy: identity}
    telefono			: {type: integer, length: 255, notnull: true}

Grupo:
  options:
    table: grupos
  fields:
    id                  : {type: primary, strategy: identity}
    nombre              : {type: string, length: 255}
    descripcion         : {type: note, length: 500}

Page:
  options:
    table: paginas
    timestampable: true
    sluggable: [title]
  fields:
    id					: {type: primary, strategy: identity}
    title				: {type: string, length: 255, notnull: true}
    orden				: {type: integer, length: 5, notnull: false}
    nombre				: {type: name, length: 30}
    email				: {type: email, length: 50, unique: true}
    login				: {type: login, length: 100, unique: true}
    password			: {type: password, length: 50, unique: true}
    url					: {type: url, length: 50}
    comentario		    : {type: editor, notnull: false}
    direccion			: {type: note, length: 2000}
    saldo				: {type: decimal, length: 10, precision: 10, scale: 2}
    fecha   			: {type: date}
    creacion			: {type: datetime}
    hora				: {type: time, notnull: true}
    tipo				: {type: options, values: {admin: Admin, normal: Normal}, default: admin, notnull: false}
    file				: {type: file, length: 50, size: 1000, ext: [jpg]}
    foto:
      type: image
      length: 50
      size: 1000
      ext: [png, jpg]
      min_dimensions: [120, 120]
      max_dimensions: [2000, 2000]
      copys:
        - {width: 400, height: 200, method: crop, dir: / }
    categoria:
      type: foreign
      model: Categoria
      join: { name: "categoria_id", foreignField: "id" }
      relation: "many-to-one"
    grupos:
      type: table
      model: Grupo
      joinTable:
        name: "pages_grupos"
        join:		{ name: "pagina_id", foreignField: "id"}
        inverseJoin : { name: "grupo_id", foreignField: "id"}
      relation: "many-to-many"
    parent:
      type: foreign
      model: Page
      join: { name: "parent_id", foreignField : "id" }
      relation: "many-to-one-self-referencing"
      nullable: yes
